            @Component({
              selector: 'my-voter',
              template: `
                
{{name}}

                
                
              `
            })
            export class VoterComponent {
              @Input()  name;
              @Output() onVoted = new EventEmitter();
              voted = false;
              vote(agreed){
                this.onVoted.emit(agreed);
                this.voted = true;
              }
            }
